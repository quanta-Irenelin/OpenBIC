46static inline void pack_ipmi_resp(struct ipmi_response *resp, ipmi_msg *ipmi_resp)
47{
48 resp->netfn = (ipmi_resp->netfn + 1) << 2; // ipmi netfn response package
49 resp->cmd = ipmi_resp->cmd;
50 resp->cmplt_code = ipmi_resp->completion_code;
51 if (ipmi_resp->data_len != 0) {
52 memcpy(resp->data, ipmi_resp->data, ipmi_resp->data_len);
53 }
54}
