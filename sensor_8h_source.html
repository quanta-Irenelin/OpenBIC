104static inline int calculate_accurate_MBR(uint8_t sensor_num, int val)
105{ // for better accuracy, enlarge SDR to two byte scale
106 if (SDR_M(sensor_num) == 0) {
107 return ((val << 8) * SDR_Rexp(sensor_num));
108 }
109 return ((val << 8) / SDR_M(sensor_num) * SDR_Rexp(sensor_num));
110}
111
112static inline int calculate_MBR(uint8_t sensor_num, int val)
113{
114 if (SDR_M(sensor_num) == 0) {
115 return (val * SDR_Rexp(sensor_num) + round_add(sensor_num, val));
116 }
117 return (val * SDR_Rexp(sensor_num) / SDR_M(sensor_num) + round_add(sensor_num, val));
118}
119
120static inline float convert_MBR_to_reading(uint8_t sensor_num, uint8_t val)
121{
122 if (SDR_M(sensor_num) == 0) {
123 return (val - round_add(sensor_num, val)) / SDR_Rexp(sensor_num);
124 }
125
126 return (val - round_add(sensor_num, val)) * SDR_M(sensor_num) / SDR_Rexp(sensor_num);
127}
